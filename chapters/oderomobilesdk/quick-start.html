Get started with the Odero Mobile SDK payment integration. This guide covers the essential backend setup required to create payment sessions.

Payment Session Creation

In your merchant backend, after you validate your merchant ID and process products, you will need to provide us a payment session ID through our payment session creation endpoint.

The payment session serves as the secure bridge between your backend validation and the mobile payment processing, ensuring that all transaction details are verified before payment initiation.

iOS Integration

Platform-specific implementation details for iOS applications using the Odero Mobile SDK.

1. Import SDK

import OderoPaySDKIOS

2. Initialize SDK

private var oderoPaySDK: OderoPaySDK? = OderoPaySDK()

// Initialize the SDK instance
oderoPaySDK = OderoPaySDK()

3. Configure Saved Cards

// Configure saved cards
let savedCards: [SavedCardItem] = [
    SavedCardItem(
        cardToken: userCardToken, 
        cvv: savedCvv, 
        lastFourDigits: cardLastFour, 
        expirationMonth: cardExpiryMonth, 
        expirationYear: cardExpiryYear
    ),
    // Add additional saved cards as needed
]

4. Show Payment Sheet

// Display the payment bottom sheet
oderoPaySDK?.showPaymentSheet(
    sessionId: paymentSessionId,
    appearance: OderoPaymentBottomSheetAppearance(
        oderoPaymentDetailsPageAppearance: OderoPaymentDetailsPageAppearance(),
        oderoPaymentProcessingPageAppearance: OderoPaymentProcessingPageAppearance(),
        oderoPaymentResultSuccessPageAppearance: OderoPaymentResultSuccessPageAppearance(),
        oderoPaymentResultFailurePageAppearance: OderoPaymentResultFailurePageAppearance(),
        oderoPaymentSummaryPageAppearance: OderoPaymentSummaryPageAppearance(),
        oderoPayWithAnotherCardPageAppearance: OderoPayWithAnotherCardPageAppearance(),
        oderoEnterCVCPageAppearance: OderoEnterCVCPageAppearance()
    ),
    applePayMerchantId: applePayMerchantIdentifier,
    savedCards: savedCards
)

// Note: Replace paymentSessionId, applePayMerchantIdentifier, and saved card variables
// with your actual values obtained from your backend and Apple Developer account.

Android Integration

Platform-specific implementation details for Android applications using the Odero Mobile SDK.

1. Initialize SDK

private var oderoPaySDK: OderoPaySdk? = null

// Initialize the SDK instance
oderoPaySDK = OderoPaySdk(
    activity = this,
    LOAD_PAYMENT_DATA_REQUEST_CODE = LOAD_PAYMENT_DATA_REQUEST_CODE,
    SCREEN_3DS_REQUEST_CODE = SCREEN_3DS_REQUEST_CODE
)

2. Configure Saved Cards

// Configure saved cards
val savedCards = listOf(
    SavedCardItem(
        cardToken = userCardToken,
        cvv = savedCvv,
        lastFourDigits = cardLastFour,
        expirationMonth = cardExpiryMonth,
        expirationYear = cardExpiryYear
    ),
    // Add additional saved cards as needed
)

3. Show Payment Sheet

// Display the payment bottom sheet
oderoPaySDK?.showPaymentSheet(
    sessionId = paymentSessionId,
    appearance = OderoPaymentBottomSheetAppearance(
        oderoPaymentDetailsPageAppearance = OderoPaymentDetailsPageAppearance(),
        oderoPaymentProcessingPageAppearance = OderoPaymentProcessingPageAppearance(),
        oderoPaymentResultSuccessPageAppearance = OderoPaymentResultSuccessPageAppearance(),
        oderoPaymentResultFailurePageAppearance = OderoPaymentResultFailurePageAppearance(),
        oderoPaymentSummaryPageAppearance = OderoPaymentSummaryPageAppearance(),
        oderoPayWithAnotherCardPageAppearance = OderoPayWithAnotherCardPageAppearance(),
        oderoEnterCVCPageAppearance = OderoEnterCVCPageAppearance()
    ),
    savedCards = savedCards
)

// Note: Replace paymentSessionId and saved card variables with your actual values
// obtained from your backend and configured request codes.

4. Handle Activity Results

// Handle Google Pay and 3DS authentication results
override fun onActivityResult(requestCode: Int, resultCode: Int, data: Intent?) {
    super.onActivityResult(requestCode, resultCode, data)
    
    when (requestCode) {
        LOAD_PAYMENT_DATA_REQUEST_CODE -> {
            oderoPaySDK?.handleGooglePayActivityResult(resultCode, data)
        }
        SCREEN_3DS_REQUEST_CODE -> {
            oderoPaySDK?.handle3DSScreenActivityResult(resultCode, data)
        }
    }
}
