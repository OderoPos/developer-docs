class SampleActivity : AppCompatActivity() {

    private val opit: Opit by lazy {
        Opit(this, lifecycleScope, opitResultListener)
    }

    //in this callback/listener you will get the result/response of the operation, from TR400
    private val opitResultListener = object : OpitResultListener() {
        override fun onPaymentResult(result: PaymentResult) {
            Debug.print("result: " + result)
        }

        override fun onVoidResult(result: VoidResult) {
            Debug.print("result: " + result)
        }

        override fun onReprintLastResult(result: ReprintReceiptLastResult) {
            Debug.print("result: " + result)
        }

        override fun onReprintReceiptNoResult(result: ReprintReceiptNumberResult) {
            Debug.print("result: " + result)
        }

        override fun onDetailedXReportResult(result: DetailedXResult) {
            Debug.print("result: " + result)
        }

        override fun onSimpleXReportResult(result: SimpleXResult) {
            Debug.print("result: " + result)
        }

        override fun onLastSettlementResult(result: LastSettlementResult) {
            Debug.print("result: " + result)
        }

        override fun onSettlementResult(result: SettlementResult) {
            Debug.print("result: " + result)
        }

        override fun onConnected() {
        }

        override fun onDisconnected() {
        }

        override fun onError(msg: String) {
            Debug.print("errorMessage: " + msg)
        }

        override fun onFiscalCommandResult(success: Boolean, result: EcrResult.FiscalResult) {
            Debug.print("result: " + result)
        }

        override fun onScreenCommandResult(success: Boolean, result: EcrResult.ScreenResult) {
            Debug.print("result: " + result)
        }
    }

    @SuppressLint("SetTextI18n")
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)

        //... other code

        opit.init()
    }

    private fun setControls(){
        buttonSetTime.setOnClickListener { sendSetTimeRequest() }
        buttonUpdateUserScreen.setOnClickListener { sendUpdateUserScreenRequest() }
        buttonSale.setOnClickListener { sendPaymentRequest() }
        buttonVoid.setOnClickListener { sendVoidRequest() }
        buttonReprintLast.setOnClickListener { reprintLastReceipt() }
        buttonReprintReceipt.setOnClickListener { sendReprintReceiptByNumber() }
        buttonDetailedX.setOnClickListener { getDetailedReport() }
        buttonSimpleX.setOnClickListener { getSimpleReport() }
        buttonLastSettlementCopy.setOnClickListener { getLastSettlementCopy() }
        buttonStartSettlement.setOnClickListener { printSettlement() }
        buttonTestAccessory.setOnClickListener { testAccessory() }
        buttonRestartConnection.setOnClickListener { restartConnection() }
        buttonIsPosConnected.setOnClickListener { isPosConnected() }
    }


    private fun sendSetTimeRequest() {
        val currentTime = Date()
        val uniqueId = UUID.randomUUID().toString()
        CoroutineScope(Dispatchers.IO).launch {
            val wasSent = opit.sendFiscalCommand(
                command = FiscalCommand.SetDateTime(currentTime),
                uniqueId = uniqueId
            )
        }
    }

    private fun sendUpdateUserScreenRequest() {
        val text = "you have to send just numbers"

        val uniqueId = UUID.randomUUID().toString()
        CoroutineScope(Dispatchers.IO).launch {
            val wasSent = opit.sendScreenCommand(text, uniqueId)
        }
    }

    private fun sendPaymentRequest() {
        val amount = 100.0
        val currency = "RON"

        val uniqueId = UUID.randomUUID().toString()
        CoroutineScope(Dispatchers.IO).launch {
            val wasSent = opit.sendPaymentRequest(amount, currency, uniqueId)
        }
    }

    private fun sendVoidRequest() {
        val receiptNo = "1234"

        val uniqueId = UUID.randomUUID().toString()
        CoroutineScope(Dispatchers.IO).launch {
        val wasSent = opit.voidReceipt(receiptNo, uniqueId)
        }
    }

    private fun reprintLastReceipt() {
        val uniqueId = UUID.randomUUID().toString()
        appendToUI("client> ReprintLastReceipt, uniqueId: $uniqueId")

        CoroutineScope(Dispatchers.IO).launch {
            val wasSent = opit.reprintLastReceipt(uniqueId)
        }
    }

    private fun sendReprintReceiptByNumber() {
        val receiptNo = "1234"

        val uniqueId = UUID.randomUUID().toString()
        CoroutineScope(Dispatchers.IO).launch {
            val wasSent = opit.reprintReceiptNo(receiptNo, uniqueId)
        }
    }

    private fun getDetailedReport() {
        val currency = "RON"

        val uniqueId = UUID.randomUUID().toString()
        CoroutineScope(Dispatchers.IO).launch {
            val wasSent = opit.getDetailedXReport(currency, uniqueId)
        }
    }

    private fun getSimpleReport() {
        val currency = "RON"

        val uniqueId = UUID.randomUUID().toString()
        CoroutineScope(Dispatchers.IO).launch {
            val wasSent = opit.getSimpleXReport(currency, uniqueId)
        }
    }

    private fun getLastSettlementCopy() {
        val currency = "RON"

        val uniqueId = UUID.randomUUID().toString()
        CoroutineScope(Dispatchers.IO).launch {
            val wasSent = opit.printLastSettlement(currency, uniqueId)
        }
    }

    private fun printSettlement() {
        val uniqueId = UUID.randomUUID().toString()
        CoroutineScope(Dispatchers.IO).launch {
            val wasSent = opit.printSettlement(uniqueId)
        }
    }

    private fun testAccessory() {
        val hasUsbAccessory = packageManager.hasSystemFeature(PackageManager.FEATURE_USB_ACCESSORY)
        val hasUsbHost = packageManager.hasSystemFeature(PackageManager.FEATURE_USB_HOST)
        val text = "Has Usb Accessory: $hasUsbAccessory\nHas Usb Host: $hasUsbHost"
        Toast.makeText(this, text, Toast.LENGTH_LONG).show()
    }

    private fun restartConnection() {
        opit.restart()
    }

    private fun isPosConnected() {
        Debug.print("isPosConnected: " + opit.isPosConnected())
    }
}
